\begin{subquestion}{\textbf{FSA Construction for $D(k,m)$}.}

\textbf{Formal Definition:}
We define the FSA $\mathcal{A} = (\Sigma, Q, \delta, I, F)$ as follows:
\begin{itemize}
    \item \textbf{$\Sigma$:} The alphabet of $2k$ brackets: $\Sigma = \{ \langle_1, \dots, \langle_k, \rangle_1, \dots, \rangle_k \}$.
    \item \textbf{$Q$:} The set of states representing all valid stack configurations up to depth $m$, using the notation $[s]$ for a stack sequence $s$:
    $Q = \{ [s] \mid s \in \{ \langle_1, \dots, \langle_k \}^* \text{ and } |s| \le m \} \cup \{ q_{rej} \}$.
    \item \textbf{$I$:} The initial state is the empty stack: $I = [\epsilon]$.
    \item \textbf{$F$:} The set of final states is $F = \{ [\epsilon] \}$, as a valid Dyck string must end with an empty stack.
    \item \textbf{$\delta$:} The transition function $\delta: Q \times \Sigma \to Q$ for any state $[s] \neq q_{rej}$:
    \begin{itemize}
        \item \textbf{Push:} For an opening bracket $\langle_n$: $\delta([s], \langle_n) = [s \cdot \langle_n]$ if $|s| < m$, otherwise $q_{rej}$.
        \item \textbf{Pop:} For a closing bracket $\rangle_n$: $\delta([s], \rangle_n) = [s']$ if $s = s' \cdot \langle_n$, otherwise $q_{rej}$.
        \item \textbf{Sink:} $\delta(q_{rej}, \sigma) = q_{rej}$ for all $\sigma \in \Sigma$.
    \end{itemize}
\end{itemize}

\textbf{Reasoning about Correctness:}
The automaton acts as a simulation of a stack. Each state $[s]$ tracks the sequence of currently unclosed brackets. 
\begin{itemize}
    \item When we see an opening bracket, we push it onto our state label. If the stack is already at depth $m$, we move to $q_{rej}$ because the string has exceeded the allowed nesting depth.
    \item When we see a closing bracket, we check if it matches the top of our state's stack. If it matches, we pop it. If it doesn't match, it's an invalid Dyck sequence, and we move to $q_{rej}$.
    \item Since we only accept at the empty stack state $[\epsilon]$, $\mathcal{A}$ correctly recognizes only those strings where every bracket was opened and closed in the correct order without ever exceeding depth $m$.
\end{itemize}

\textbf{Complexity Analysis:}
\begin{itemize}
    \item \textbf{States:} The number of possible stack sequences of length $i$ is $k^i$. Summing from length 0 to $m$ gives $\sum_{i=0}^m k^i$. Thus, the number of states $|Q|$ is $\mathcal{O}(k^m)$.
    \item \textbf{Transitions:} Every state has $2k$ outgoing transitions (one for each bracket type). The total number of transitions is $\mathcal{O}(k \cdot k^m) = \mathcal{O}(k^{m+1})$.
\end{itemize}

\end{subquestion}

\begin{subquestion}{\textbf{Flexible RNN Model.}}

We define the RNN dynamics $h_t = U(y_t)h_{t-1} + Vo_{y_t}$ such that the hidden state $h_t \in \mathbb{R}^{km}$ tracks the stack configuration.

\begin{itemize}
    \item \textbf{Recurrence Matrix $U(y_t) \in \mathbb{R}^{km \times km}$:} 
    Depending on whether the input $y_t$ pushes to or pops from the stack, we define:
    \[
    U(y_t) = \begin{cases} 
        U_{\text{push}} & \text{if } y_t \in \{ \langle_1, \dots, \langle_k \} \\
        U_{\text{pop}} & \text{if } y_t \in \{ \rangle_1, \dots, \rangle_k \} \\
        \mathbf{I}_{km} & \text{if } y_t = \text{EOS}
    \end{cases}
    \]
    where the block-shift matrices are defined as:
    \[
    U_{\text{push}} = \begin{pmatrix}
        \mathbf{0}_{k \times k} & \mathbf{0}_{k \times (m-1)k} \\
        \mathbf{I}_{(m-1)k} & \mathbf{0}_{(m-1)k \times k}
    \end{pmatrix}, \quad
    U_{\text{pop}} = \begin{pmatrix}
        \mathbf{0}_{(m-1)k \times k} & \mathbf{I}_{(m-1)k} \\
        \mathbf{0}_{k \times k} & \mathbf{0}_{k \times (m-1)k}
    \end{pmatrix}
    \]

    \item \textbf{Input Matrix $V \in \mathbb{R}^{km \times (2k+1)}$:} 
    $V$ extracts the one-hot encoding of an opening bracket and places it at the top of the stack:
    \[
    V = \begin{pmatrix}
        \mathbf{I}_{k} & \mathbf{0}_{k \times k} & \mathbf{0}_{k \times 1} \\
        \mathbf{0}_{(m-1)k \times k} & \mathbf{0}_{(m-1)k \times k} & \mathbf{0}_{(m-1)k \times 1}
    \end{pmatrix}
    \]
\end{itemize}

\textbf{Correspondence to Automaton $\mathcal{A}$:}
\begin{itemize}
    \item \textbf{Push:} When $y_t \in \{ \langle_1, \dots, \langle_k \}$, $U_{\text{push}}$ shifts the existing stack down to make room at the top. $V$ inserts the new bracket $e_n$ into that top slot. This results in a hidden state $h_t = (e_n, s_1, \dots, s_{m-1})^\top$, matching the state $[s \cdot \langle_n]$.
    \item \textbf{Pop:} When $y_t \in \{ \rangle_1, \dots, \rangle_k \}$, $V$ contributes a zero vector. $U_{\text{pop}}$ shifts the existing stack up by one position, and promoting the element below it. This simulates the transition $[s' \cdot \langle_n] \to [s']$ in the hidden state space.
\end{itemize}

\end{subquestion}

\begin{subquestion}{\textbf{Standard Elman RNN.}}

We define the parameters for the Elman RNN $h_t = H(Uh_{t-1} + Vo_{y_t} + b)$ to simulate the bounded stack automaton $\mathcal{A}$.

\begin{itemize}
    \item \textbf{Hidden State $h_t \in \mathbb{R}^{2km}$:} 
    We partitioned the state into two tracks of size $km$, the top "push" track and a bottom "pop" track. At any time, only one track is active (encoding the stack state) while the other remains dormant (all zeros).

    \item \textbf{Fixed Recurrence Matrix $U \in \mathbb{R}^{2km \times 2km}$:} 
    The matrix $U$ applies the shift operations to both tracks. It draws from whichever track was active in to prepare the next state's candidates:
    \[
    U = \begin{pmatrix}
        U_{\text{push}} & U_{\text{push}} \\
        U_{\text{pop}} & U_{\text{pop}}
    \end{pmatrix}
    \]
    where $U_{\text{push}}$ and $U_{\text{pop}}$ are the block-shift matrices defined in 3d.

    \item \textbf{Bias $b$ and Input Matrix $V$:} 
    With $b = -\mathbf{1}_{2km}$, the matrix $V$ provides a $+1$ "boost" to activate the correct track based on the symbol order in Eq. (28):
    \[
    V = \begin{pmatrix}
        \mathbf{I}_{k} + \mathbf{1}_{k \times k} & \mathbf{0}_{k \times k} & \mathbf{1}_{k \times 1} \\
        \mathbf{1}_{(m-1)k \times k} & \mathbf{0}_{(m-1)k \times k} & \mathbf{1}_{(m-1)k \times 1} \\
        \mathbf{0}_{k \times k} & \mathbf{1}_{k \times k} & \mathbf{1}_{k \times 1} \\
        \mathbf{0}_{(m-1)k \times k} & \mathbf{1}_{(m-1)k \times k} & \mathbf{1}_{(m-1)k \times 1}
    \end{pmatrix}
    \]

\end{itemize}

\textbf{Correspondence to Automaton $\mathcal{A}$:} \\
The activation $H$ acts as a selector by only promoting entries that receive a boost from $V$:
\begin{itemize}
    \item \textbf{Push ($y_t \in \{\langle_1 \dots \langle_k\}$):} $V$ boosts the top track to activate the transition $[s \cdot \langle_n]$, while the bottom track remains at $-1$ and is zeroed.
    \item \textbf{Pop ($y_t \in \{\rangle_1 \dots \rangle_k\}$):} $V$ boosts the bottom track to activate state $[s']$, suppressing the push track.
    \item \textbf{EOS:} $V$ boosts both tracks to preserve the final stack configuration in the hidden state.
\end{itemize}

\end{subquestion}

\textbf{Part III. Recognition with an RNN}

\begin{subquestion}{\textbf{Formal Recognition.}}

To decode the $2km$-dimensional hidden state from 3e, we define the output matrix $E$ and bias vector $b'$ to compute the logits $Eh_{t-1} + b'$. We use the redundant block structure $E = \begin{pmatrix} E_{\text{track}} & E_{\text{track}} \end{pmatrix}$ since only one track of $h_{t-1}$ is active at a time. Using a real-valued parameter $\alpha > 0$, we define:

\[
b' = \begin{pmatrix} \alpha \cdot \mathbf{1}_k \\ \mathbf{0}_k \\ \alpha \end{pmatrix}, \quad 
E_{\text{track}} = \begin{pmatrix} 
\mathbf{0}_{k \times (m-1)k} & -\alpha \cdot \mathbf{1}_{k \times k} \\
\alpha \cdot \mathbf{I}_k & \mathbf{0}_{k \times (m-1)k} \\
-\alpha \cdot \mathbf{1}_{1 \times mk}
\end{pmatrix}
\]

\textbf{Logit Reasoning for Support $\mathcal{L}_{\eta}(p_{\text{SM}})$:} \\
The construction ensures logits $z_j \approx \alpha$ for valid continuations and $z_j \le 0$ for invalid ones:
\begin{itemize}
    \item \textbf{Opening Brackets:} Use \textit{Veto Logic}. $b'$ grants default permission ($\alpha$), but $E_{\text{track}}$ subtracts $\alpha$ if the $m$-th stack slot is occupied, rejecting the symbol when the depth limit is reached.
    \item \textbf{Closing Brackets:} Use \textit{Permission Logic}. $b'$ is $0$ (default rejection). $E_{\text{track}}$ adds $\alpha$ only if the corresponding opening bracket $i$ matches the stack top (Slot 1).
    \item \textbf{EOS:} Uses \textit{Veto Logic}. $b'$ is $\alpha$, but $E_{\text{track}}$ subtracts $\alpha$ if any bracket exists in the stack, ensuring termination only for empty stacks.
\end{itemize}

\textbf{Bounding the Parameter $\alpha$:} \\
To ensure $p(y_t | y_{<t}) \ge \eta$ for acceptable symbols, we analyze the normalization constant $Z = \sum_{j} \exp(z_j)$. There are three relevant cases for $Z$:
\begin{enumerate}
    \item \textbf{Empty Stack ($d=0$):} $k$ opening brackets and EOS are valid. $Z_0 = (k+1)e^{\alpha} + k \cdot e^0$.
    \item \textbf{Partial Stack ($1 \le d < m$):} $k$ opening brackets and $1$ matching closing bracket are valid. $Z_{\text{partial}} = (k+1)e^{\alpha} + (k-1) \cdot e^0 + e^{\alpha(1-d)}$.
    \item \textbf{Full Stack ($d=m$):} Only $1$ matching closing bracket is valid. $Z_{\text{full}} = e^{\alpha} + 2k \cdot e^0$.
\end{enumerate}
The worst-case (largest $Z$) occurs when $d=0$ or $d=1$, where $Z_{\text{max}} = (k+1)e^{\alpha} + k$. To satisfy the $\eta$-truncated support requirement:
\[ \frac{e^{\alpha}}{(k+1)e^{\alpha} + k} \ge \eta \implies e^{\alpha}(1 - \eta(k+1)) \ge \eta k \]
Given $\eta < \frac{1}{k+1}$, the lower bound for $\alpha$ as a function of $\eta$ is:
\[ \alpha \ge \ln \left( \frac{\eta k}{1 - \eta(k+1)} \right) \]

\textbf{Conclusion:} By setting $\alpha$ sufficiently high, all invalid continuations yield probabilities $p < \eta$, while valid ones maintain $p \ge \eta$, such that $\mathcal{L}_{\eta}(p_{\text{SM}}) = D(k,m)$.

\end{subquestion}

\begin{subquestion}{\textbf{RNN Parameters with Binary Encoding.}}

Each bracket is assigned a unique binary encoding of length $L = \lceil \log_2 k \rceil + 1$. Maintaining the two-track logic from question (e), the hidden state dimension becomes $2Lm$.

\textbf{Dimensions:}
\begin{itemize}
    \item $U \in \mathbb{R}^{2Lm \times 2Lm}$
    \item $V \in \mathbb{R}^{2Lm \times (2k+1)}$
    \item $b \in \mathbb{R}^{2Lm}$
\end{itemize}

\textbf{Parameter Changes:}
\begin{itemize}
    \item \textbf{Recurrence Matrix $U$}: The internal block-shift size used for push and pop operations  changes from $k \times k$ to $L \times L$.
    \item \textbf{Input Matrix $V$}: The $k$ columns corresponding to opening brackets now encode the specific $L$-bit binary pattern $e_n$ for each bracket into the top slot of the push track, rather than acting as a sparse identity block.
    \item \textbf{Bias $b$}: The vector length is reduced to $2Lm$ to match the compressed hidden state.
\end{itemize}

\end{subquestion}

\begin{subquestion}{\textbf{Formal Recognition with Binary Encoding}.}

We define the output matrix $E = \begin{pmatrix} E_{\text{track}} & E_{\text{track}} \end{pmatrix} \in \mathbb{R}^{(2k+1) \times 2Lm}$ to decode the binary hidden state. Each track $E_{\text{track}} \in \mathbb{R}^{(2k+1) \times Lm}$ maps the $m$ stack slots of $L = \lceil \log_2 k \rceil + 1$ bits to symbol logits.

Let $e_n \in \{0, 1\}^L$ be the binary representation of bracket type $n$. We define $|e_n|$ as the number of set bits (1s) in that encoding. Using a parameter $\alpha > 0$, the bias $b' \in \mathbb{R}^{2k+1}$ and matrix $E_{\text{track}}$ are defined as:

\[
b' = \begin{pmatrix} 
\alpha \cdot \mathbf{1}_k \\ 
\alpha(1 - |e_1|, \dots, 1 - |e_k|)^\top \\ 
\alpha 
\end{pmatrix}, \quad 
E_{\text{track}} = \begin{pmatrix} 
\mathbf{0}_{k \times (m-1)L} & -\alpha \cdot \mathbf{1}_{k \times L} \\
\alpha \cdot P & \mathbf{0}_{k \times (m-1)L} \\
-\alpha \cdot \mathbf{1}_{1 \times mL}
\end{pmatrix}
\]

Where $P \in \{1, -1\}^{k \times L}$ is a pattern matrix such that $P_{n,i} = 1$ if the $i$-th bit of $e_n$ is 1, and $P_{n,i} = -1$ otherwise.

\paragraph{Logit Reasoning for Support $\mathcal{L}_{\eta}(p_{\text{SM}})$:}
Following the symbol order in Eq. (28), this construction ensures logits $z_j = \alpha$ for valid symbols and $z_j \le 0$ for invalid ones:
\begin{itemize}
    \item \textbf{Opening Brackets ($\langle_1, \dots, \langle_k$):} Uses \textbf{Veto Logic}. The bias grants default permission ($\alpha$). If any bit in the $m$-th stack slot is 1 (depth limit reached), the $-\alpha$ terms in $E_{\text{track}}$ cancel the permission.
    \item \textbf{Closing Brackets ($\rangle_1, \dots, \rangle_k$):} Uses \textbf{Pattern Matching Logic}. The logit for bracket $n$ is $z = \alpha(1 - |e_n|) + \alpha(\text{matched bits} - \text{mismatched bits})$. If the top slot perfectly matches $e_n$, the logit is exactly $\alpha$. Any mismatch results in $z \le 0$.
    \item \textbf{EOS:} Uses \textbf{Veto Logic}. The bias is $\alpha$. If any bit in any of the $m$ stack slots is 1, the $-\alpha$ entries in $E_{\text{track}}$ veto the logit, ensuring the string terminates only on an empty stack.
\end{itemize}

\end{subquestion}
